<body>
  <h1>Blackjack Online Advanced</h1>
  <input type="text" id="player_name" placeholder="Enter name">
  <input type="number" id="bet_amount" placeholder="Bet amount" value="10">
  <button onclick="join()">Join & Deal</button>
  <button onclick="hit()">Hit</button>
  <button onclick="stand()">Stand</button>
  <button onclick="split()">Split</button>

  <pre id="output"></pre>

  <script>
    var socket = io();
    var playerName = "";

    function join(){
      playerName = document.getElementById("player_name").value || "Player";
      var bet = parseInt(document.getElementById("bet_amount").value) || 10;
      socket.emit("join",{name:playerName, bet:bet});
    }

    function hit(split=false){
      socket.emit("hit",{name:playerName, split:split});
    }

    function stand(split=false){
      socket.emit("stand",{name:playerName, split:split});
    }

    function split(){
      socket.emit("split",{name:playerName});
    }

    socket.on("game_state", function(data){
      document.getElementById("output").innerText = JSON.stringify(data,null,2);
    });
  </script>
